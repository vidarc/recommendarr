> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Enable users to connect Radarr/Sonarr and add AI-recommended media to those services from recommendation cards.

**Architecture:** A shared `arr-client.ts` service abstracts both Radarr and Sonarr v3 APIs behind a common interface. New Fastify routes handle connection CRUD, testing, lookup, and add operations. The frontend gets a working IntegrationsTab in settings and an AddToArrModal triggered from RecommendationCard buttons.

**Tech Stack:** Fastify, Drizzle ORM, Zod, RTK Query, React, Linaria, MSW (testing), Playwright (e2e)

const ARR_API_PREFIX = "/api/v3";

interface TestConnectionResult {
	success: boolean;
	version?: string;
	error?: string;
}

const arrFetch = async (url: string, apiKey: string, path: string, init?: RequestInit) =>
	fetch(`${url}${ARR_API_PREFIX}${path}`, {
		...init,
		headers: { "X-Api-Key": apiKey, ...init?.headers },
	});

const testConnection = async (url: string, apiKey: string): Promise<TestConnectionResult> => {
	try {
		const response = await arrFetch(url, apiKey, "/system/status");
		if (!response.ok) {
			return { success: false, error: `HTTP ${String(response.status)}` };
		}
		const data = (await response.json()) as { version?: string };
		return { success: true, version: data.version };
	} catch (error) {
		return { success: false, error: error instanceof Error ? error.message : "Unknown error" };
	}
};

export { testConnection };
export type { TestConnectionResult };
// NOTE: As more functions are added, consolidate into a single export { ... } and export type { ... }
// per the project's import/group-exports lint rule.

- [ ] **Step 9: Implement `lookupMedia`**

Add to `arr-client.ts`:

```typescript
interface LookupResult {
	title: string;
	year: number;
	tmdbId?: number;
	tvdbId?: number;
	overview: string;
	existsInLibrary: boolean;
	arrId: number;
}

type ServiceType = "radarr" | "sonarr";

const lookupMedia = async (
	url: string,
	apiKey: string,
	serviceType: ServiceType,
	title: string,
	year?: number,
): Promise<LookupResult[]> => {
	const term = year ? `${title} ${String(year)}` : title;
	const lookupPath = serviceType === "radarr" ? "/movie/lookup" : "/series/lookup";
	const response = await arrFetch(url, apiKey, `${lookupPath}?term=${encodeURIComponent(term)}`);

	if (!response.ok) {
		throw new Error(`Lookup failed: HTTP ${String(response.status)}`);
	}

	const results = (await response.json()) as Array<{
		title: string;
		year: number;
		tmdbId?: number;
		tvdbId?: number;
		overview: string;
		hasFile?: boolean;
		id: number;
	}>;

	return results.map((item) => ({
		title: item.title,
		year: item.year,
		tmdbId: item.tmdbId,
		tvdbId: item.tvdbId,
		overview: item.overview,
		existsInLibrary: item.id > 0,
		arrId: item.id,
	}));
};
```

Note: Both Radarr and Sonarr return `id: 0` for items NOT in the library and a real ID (`id > 0`) for items already added. The `hasFile` field on Radarr indicates download status, not library presence — a movie can be in the library but not yet downloaded. Using `id > 0` is the correct check for both services.

```typescript
interface AddMediaParams {
	tmdbId?: number;
	tvdbId?: number;
	title: string;
	year: number;
	qualityProfileId: number;
	rootFolderPath: string;
}

interface AddMediaResult {
	success: boolean;
	id?: number;
	error?: string;
}

const addMedia = async (
	url: string,
	apiKey: string,
	serviceType: ServiceType,
	params: AddMediaParams,
): Promise<AddMediaResult> => {
	const path = serviceType === "radarr" ? "/movie" : "/series";

	const body =
		serviceType === "radarr"
			? {
					tmdbId: params.tmdbId,
					title: params.title,
					year: params.year,
					qualityProfileId: params.qualityProfileId,
					rootFolderPath: params.rootFolderPath,
					minimumAvailability: "released",
					monitored: true,
					addOptions: { searchForMovie: true },
				}
			: {
					tvdbId: params.tvdbId,
					title: params.title,
					year: params.year,
					qualityProfileId: params.qualityProfileId,
					rootFolderPath: params.rootFolderPath,
					monitored: true,
					seasonFolder: true,
					seriesType: "standard",
					addOptions: { searchForMissingEpisodes: true },
				};

	try {
		const response = await arrFetch(url, apiKey, path, {
			method: "POST",
			headers: { "Content-Type": "application/json" },
			body: JSON.stringify(body),
		});

		if (!response.ok) {
			const errorBody = await response.text();
			return { success: false, error: `HTTP ${String(response.status)}: ${errorBody}` };
		}

		const data = (await response.json()) as { id: number };
		return { success: true, id: data.id };
	} catch (error) {
		return { success: false, error: error instanceof Error ? error.message : "Unknown error" };
	}
};
```

```typescript
import { useCallback, useEffect, useState } from "react";

import {
	useDeleteArrConfigMutation,
	useGetArrConfigQuery,
	useTestArrConnectionMutation,
	useUpdateArrConfigMutation,
} from "../api.ts";

import type { ArrConnection } from "../api.ts";

type ServiceType = "radarr" | "sonarr";

interface ServiceState {
	url: string;
	apiKey: string;
}

const useArrConfig = () => {
	const { data: connections } = useGetArrConfigQuery();
	const [updateConfig] = useUpdateArrConfigMutation();
	const [deleteConfig] = useDeleteArrConfigMutation();
	const [testConnection] = useTestArrConnectionMutation();

	const [radarr, setRadarr] = useState<ServiceState>({ url: "", apiKey: "" });
	const [sonarr, setSonarr] = useState<ServiceState>({ url: "", apiKey: "" });
	const [testResult, setTestResult] = useState<Record<string, string>>({});

	useEffect(() => {
		if (connections) {
			for (const conn of connections) {
				const state = { url: conn.url, apiKey: conn.apiKey };
				if (conn.serviceType === "radarr") setRadarr(state);
				if (conn.serviceType === "sonarr") setSonarr(state);
			}
		}
	}, [connections]);

	const handleSave = useCallback(
		async (serviceType: ServiceType) => {
			const state = serviceType === "radarr" ? radarr : sonarr;
			await updateConfig({ serviceType, ...state });
		},
		[radarr, sonarr, updateConfig],
	);

	const handleTest = useCallback(
		async (serviceType: ServiceType) => {
			const result = await testConnection({ serviceType });
			if ("data" in result) {
				setTestResult((prev) => ({
					...prev,
					[serviceType]: result.data.success
						? `Connection successful (v${result.data.version ?? "unknown"})`
						: `Connection failed: ${result.data.error ?? "Unknown error"}`,
				}));
			}
		},
		[testConnection],
	);

	const handleRemove = useCallback(
		async (serviceType: ServiceType) => {
			await deleteConfig(serviceType);
			if (serviceType === "radarr") setRadarr({ url: "", apiKey: "" });
			if (serviceType === "sonarr") setSonarr({ url: "", apiKey: "" });
			setTestResult((prev) => {
				const next = { ...prev };
				delete next[serviceType];
				return next;
			});
		},
		[deleteConfig],
	);

	const isConnected = useCallback(
		(serviceType: ServiceType) =>
			connections?.some((c: ArrConnection) => c.serviceType === serviceType) ?? false,
		[connections],
	);

	return {
		radarr,
		sonarr,
		testResult,
		setRadarr,
		setSonarr,
		handleSave,
		handleTest,
		handleRemove,
		isConnected,
	};
};

export { useArrConfig };
```
